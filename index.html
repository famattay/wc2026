const STORAGE_KEY = "panini_fwc_2026_pipe_v1";


function buildInitialState() {

  const state = {};

  TEAM_ORDER.forEach(team => {

    state[team] = {};

    numbersFor(team).forEach(n => {

      const falta = (INITIAL_FALTAN[team] || []).includes(n);

      const repArr = (INITIAL_REPETIDAS[team] || []);

      const rep = repArr.includes(n);

      state[team][n] = {

        status: falta ? 0 : (rep ? 2 : 1),

        repCount: rep ? 1 : 0

      };

    });

  });

  return state;

}


function loadState() {

  try {

    const raw = localStorage.getItem(STORAGE_KEY);

  } catch(e) {}

  return buildInitialState();

}

function saveState() {

  localStorage.setItem(STORAGE_KEY, JSON.stringify(state));

}

function teamStats(team) {

  const nums = numbersFor(team);

  let have = 0, falt = 0, repUnits = 0, foilHave = 0, foilTotal = 0;

  nums.forEach(n => {

    const c = state[team][n];

    if (isFoil(team, n)) {

      foilTotal++;

      if (c.status >= 1) foilHave++;

    }

    if (c.status === 0) falt++;

    else have++;

    if (c.status === 2) repUnits += c.repCount;

  });

  return {

    total: nums.length, have, falt, repUnits, foilHave, foilTotal,

    pct: nums.length ? Math.round(have / nums.length * 100) : 0

  };

}

  // Función para mostrar tooltip (funciona en mouse y touch)

  function showTooltip(event, d) {

    const info = isoData[d.id];

    if (!info) { tooltip.classList.remove("show"); return; }

    const teamsHtml = info.teams.map(t => `

      <div class="tt-team">${t.code} · ${TEAM_NAMES[t.code]}</div>

      <div class="tt-row"><span>Tengo</span><strong style="color:var(--pitch-2)">${t.have} / ${t.total}</strong></div>

      <div class="tt-row"><span>Faltan</span><strong style="color:var(--red-2)">${t.falt}</strong></div>

      <div class="tt-row"><span>Repetidas</span><strong>${t.repUnits}</strong></div>

      <div class="tt-row"><span>Brillantes</span><strong style="color:var(--gold-2)">${t.foilHave}/${t.foilTotal}</strong></div>

      <div class="tt-pct">${t.pct}% completado</div>

    `).join('<div style="height:8px;"></div>');

    tooltip.innerHTML = teamsHtml;

    tooltip.classList.add("show");

    const rect = wrap.getBoundingClientRect();

    // Coordenada del puntero (mouse o touch)

    const px = event.touches ? event.touches[0].clientX : event.clientX;

    const py = event.touches ? event.touches[0].clientY : event.clientY;

    const ttW = tooltip.offsetWidth || 200;

    const ttH = tooltip.offsetHeight || 180;

    let x = px - rect.left + 14;

    let y = py - rect.top + 14;

    // Mantener tooltip dentro del contenedor

    if (x + ttW > rect.width) x = px - rect.left - ttW - 14;

    if (y + ttH > rect.height) y = py - rect.top - ttH - 14;

    x = Math.max(8, x);

    y = Math.max(8, y);

    tooltip.style.left = x + "px";

    tooltip.style.top = y + "px";

  }


  countryPaths

    .on("mousemove", showTooltip)

    .on("mouseleave", () => tooltip.classList.remove("show"))

    .on("touchstart", function(event, d) {

      event.preventDefault();

      showTooltip(event, d);

    }, { passive: false })

    .on("click", showTooltip);


  // Tocar fuera del mapa cierra el tooltip

  document.addEventListener("touchstart", (e) => {

    if (!wrap.contains(e.target)) tooltip.classList.remove("show");

  });


  // Etiquetas: código + tengo/faltan

  const labelG = svg.append("g");

  countries.features.forEach(d => {

    const info = isoData[d.id];

    if (!info) return;

    const centroid = path.centroid(d);

    if (isNaN(centroid[0])) return;


    info.teams.forEach((t, idx) => {

      const offsetY = idx * 11; // si hay 2 equipos en el mismo país (UK), apilar

      // Texto principal: código

      labelG.append("text")

        .attr("class", "label")

        .attr("x", centroid[0])

        .attr("y", centroid[1] - 3 + offsetY)

        .text(t.code);

      // Texto secundario: tengo/faltan

      labelG.append("text")

        .attr("class", "label small")

        .attr("x", centroid[0])

        .attr("y", centroid[1] + 6 + offsetY)

        .attr("fill", "#b8efc8")

        .text(`${t.have}/${t.falt}`);

    });

  });


  mapDrawn = true;

}


function buildIsoData() {

  const iso = {};

  Object.entries(PANINI_TO_ISO_NUM).forEach(([code, isoNum]) => {

    const s = teamStats(code);

    const teamInfo = { code, ...s };

    if (!iso[isoNum]) {

      iso[isoNum] = { pct: s.pct, teams: [teamInfo] };

    } else {

      // Caso UK (ENG + SCO): promediar % y guardar ambos

      iso[isoNum].teams.push(teamInfo);

      const avg = iso[isoNum].teams.reduce((a,b)=>a+b.pct, 0) / iso[isoNum].teams.length;

      iso[isoNum].pct = avg;

    }

  });

  return iso;

}

    type: 'bar',

    data: {

      labels: confLabels,

      datasets: [{

        label: 'Faltantes',

        data: confLabels.map(c => confData[c]),

        backgroundColor: ['#d9534f','#e0a64a','#4a8fc4','#1f9d55','#d4a437','#9d6cd9'],

        borderWidth: 0

      }]

      indexAxis: 'y',

      plugins: { legend: { display: false } },

      scales: {

        x: { ticks:{ color:'#b8c2bb', font:{family:'JetBrains Mono', size:11}}, grid:{color:'#243029'}, beginAtZero:true},

        y: { ticks:{ color:'#b8c2bb', font:{family:'JetBrains Mono', size:11}}, grid:{display:false}}

      }

    }

  });


  // Repetidas por equipo (top 15)
